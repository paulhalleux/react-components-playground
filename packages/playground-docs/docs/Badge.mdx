# Badge

`Badge` is a component to display a badge. It can be used to display a status, a count, or any other information.

```tsx
import { Badge } from "@paulhalleux/react-playground";

const GroupStyle = {
  display: "flex",
  gap: 24,
  alignItems: "center",
};

export function App() {
  return (
    <div>
      <div style={GroupStyle}>
        <Badge size="small">Default</Badge>
        <Badge size="small" variant="primary">
          Primary
        </Badge>
        <Badge size="small" variant="secondary">
          Secondary
        </Badge>
      </div>
      <div style={GroupStyle}>
        <Badge size="medium">Default</Badge>
        <Badge size="medium" variant="primary">
          Primary
        </Badge>
        <Badge size="medium" variant="secondary">
          Secondary
        </Badge>
      </div>
      <div style={GroupStyle}>
        <Badge size="large">Default</Badge>
        <Badge size="large" variant="primary">
          Primary
        </Badge>
        <Badge size="large" variant="secondary">
          Secondary
        </Badge>
      </div>
    </Display>
  );
}
```
