## API

### \{:component} Tree

It is a container for `TreeNode` components.

#### Usage

```tsx
<Tree>
  {/* ... */}
</Tree>
```

#### Properties

<Properties component="Tree" />

### \{:component} Tree.Node

Represents a node in the tree.

#### Usage

```tsx
<Tree>
  <Tree.Node id="node-1" label="Node 1">
    {/* Sub nodes */}
  </Tree.Node>
</Tree>
```

<Properties component="Tree.Node" />

type TreeNodeAddonVisibility = 'always' | 'hover';
